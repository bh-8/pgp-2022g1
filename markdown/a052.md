4. Gegeben seien folgende Funktionen in Scala. Geben Sie an, ob die Funktionen jeweils rekursiv, linear rekursiv und/oder endrekursiv sind.
```scala
def func1(n: Int): Int = {
   if (n > 100) {
      n - 10
   } else {
      func1(func1(n + 11))
   }
}

def func2(n: Int): Int = {
   if (n == 0) {
      0
   } else if (n > 0) {
      func2(n - 1) - n
   } else {
      func2(n + 1) + n
   }
}

def func3(n: Int, m: Int): Int = {
   if (n <= 0 | m <= 0) {
      n + m
   } else if (n > m) {
      func3(n - 1, m)
   } else {
      func3(n, m - 1)
   }
}

def func4A(n: Int, m: Int): Int = {
   if (n >= 0) {
      func4B(n - 1, m + 1)
   } else {
      m
   }
}
def func4B(n: Int, m: Int): Int = {
   if (n >= 0) {
      func4A(n - 2, m + 1)
   } else {
      m
   }
}
```
